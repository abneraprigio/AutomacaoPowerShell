# --- Remove arquivo/pasta agressivamente (3 estratégias em cascata) ---
function Force-RemoveItem {
    param([string]$Path)

    if (-not (Test-Path -LiteralPath $Path)) { return $true }

    try {
        Get-ChildItem -LiteralPath $Path -Recurse -Force -ErrorAction SilentlyContinue |
            ForEach-Object { try { $_.Attributes = 'Normal' } catch {} }

        $item = Get-Item -LiteralPath $Path -Force -ErrorAction SilentlyContinue
        if ($item) { try { $item.Attributes = 'Normal' } catch {} }

        Remove-Item -LiteralPath $Path -Recurse -Force -ErrorAction Stop
        $Script:TotalDeleted++
        return $true
    } catch {
        try {
            if ((Get-Item -LiteralPath $Path -Force).PSIsContainer) {
                [System.IO.Directory]::Delete($Path, $true)
            } else {
                [System.IO.File]::Delete($Path)
            }
            $Script:TotalDeleted++
            return $true
        } catch {
            try {
                if ((Get-Item -LiteralPath $Path -Force).PSIsContainer) {
                    cmd /c "rd /s /q `"$Path`"" 2>$null | Out-Null
                } else {
                    cmd /c "del /f /q `"$Path`"" 2>$null | Out-Null
                }
                if (-not (Test-Path -LiteralPath $Path)) {
                    $Script:TotalDeleted++
                    return $true
                }
            } catch {}
            $Script:TotalFailed++
            return $false
        }
    }
}

function Remove-Paths {
    param([string[]]$Paths, [switch]$Quiet)
    foreach ($p in $Paths) {
        if ($p -match '[*?]') {
            $resolved = Resolve-Path -Path $p -ErrorAction SilentlyContinue
            foreach ($r in $resolved) {
                if (Force-RemoveItem -Path $r.Path) {
                    if (-not $Quiet) { Write-Host "   [DEL] $($r.Path)" -ForegroundColor DarkGray }
                }
            }
        } else {
            if (Test-Path -LiteralPath $p) {
                if (Force-RemoveItem -Path $p) {
                    if (-not $Quiet) { Write-Host "   [DEL] $p" -ForegroundColor DarkGray }
                }
            }
        }
    }
}

function Stop-ProcessSafe {
    param([string]$ProcessName)
    $procs = Get-Process -Name $ProcessName -ErrorAction SilentlyContinue
    if ($procs) {
        taskkill /F /T /IM "$ProcessName.exe" 2>$null | Out-Null
        Write-Host "   [KILL] $ProcessName" -ForegroundColor DarkGray
    }
}

# Mata processos por padrão (wildcard) — pega TODAS as variantes
function Stop-ProcessByPattern {
    param([string]$Pattern)
    $procs = Get-Process | Where-Object { $_.Name -like $Pattern } -ErrorAction SilentlyContinue
    foreach ($p in $procs) {
        try {
            Stop-Process -Id $p.Id -Force -ErrorAction SilentlyContinue
            Write-Host "   [KILL] $($p.Name) (PID $($p.Id))" -ForegroundColor DarkGray
        } catch {}
    }
}

function Remove-RegistryKey {
    param([string]$KeyPath)
    if (Test-Path $KeyPath) {
        try {
            Remove-Item -Path $KeyPath -Recurse -Force -ErrorAction Stop
            Write-Host "   [DEL REG] $KeyPath" -ForegroundColor DarkGray
        } catch {
            Write-Host "   [FAIL REG] $KeyPath" -ForegroundColor DarkRed
        }
    }
}

function Clear-FolderContents {
    param(
        [string]$FolderPath,
        [string]$Label,
        [string[]]$ProtectedExtensions = @()
    )

    Write-Host " -> $Label" -ForegroundColor White

    if ([string]::IsNullOrWhiteSpace($FolderPath) -or -not (Test-Path -LiteralPath $FolderPath)) {
        Write-Host "   [SKIP] Pasta não encontrada." -ForegroundColor DarkYellow
        return
    }

    Write-Host "   Caminho real: $FolderPath" -ForegroundColor DarkGray

    $items = @(Get-ChildItem -LiteralPath $FolderPath -Force -ErrorAction SilentlyContinue)

    if ($items.Count -eq 0) {
        Write-Host "   [OK] Já está vazia." -ForegroundColor DarkGreen
        return
    }

    $deleted = 0; $kept = 0; $failed = 0

    foreach ($item in $items) {
        if ($item.Name -ieq "desktop.ini") { $kept++; continue }

        if (-not $item.PSIsContainer -and $ProtectedExtensions.Count -gt 0) {
            $shouldProtect = $false
            foreach ($ext in $ProtectedExtensions) {
                if ($item.Extension -ieq $ext) { $shouldProtect = $true; break }
            }
            if ($shouldProtect) {
                Write-Host "   [KEEP] $($item.Name)" -ForegroundColor DarkCyan
                $kept++; continue
            }
        }

        if (Force-RemoveItem -Path $item.FullName) { $deleted++ }
        else { $failed++; Write-Host "   [FAIL] $($item.Name)" -ForegroundColor Red }
    }

    Write-Host "   [OK] $Label — Removidos: $deleted | Preservados: $kept | Falhas: $failed" -ForegroundColor Green
}

    "Downloads (legado)"  = "$TargetProfile\Downloads"
    "Documents (legado)"  = "$TargetProfile\Documents"
    "Pictures (legado)"   = "$TargetProfile\Pictures"
    "Videos (legado)"     = "$TargetProfile\Videos"
    "Music (legado)"      = "$TargetProfile\Music"
}
foreach ($entry in $LegacyPaths.GetEnumerator()) {
    $alreadyCleaned = @($PathDesktop, $PathDownloads, $PathDocuments,
                        $PathPictures, $PathVideos, $PathMusic) -contains $entry.Value
    if (-not $alreadyCleaned -and (Test-Path -LiteralPath $entry.Value)) {
        Clear-FolderContents -FolderPath $entry.Value -Label $entry.Key
    }
}

Write-Host " [OK] Pastas pessoais limpas." -ForegroundColor Green

# ==============================================================================
# BLOCO 11 — UWP + HIGIENE COMPLEMENTAR
# ==============================================================================

Write-Host ""
Write-Host "[COMPLEMENTAR] UWP + Temp + MRU..." -ForegroundColor Yellow

# Bloco de Notas Win11
if (Test-Path "$TargetProfile\AppData\Local\Packages") {
    Get-ChildItem -Path "$TargetProfile\AppData\Local\Packages" -Directory `
        -Filter "*WindowsNotepad*" -ErrorAction SilentlyContinue |
        ForEach-Object {
            Remove-Paths -Paths @(
                "$($_.FullName)\LocalState", "$($_.FullName)\Settings",
                "$($_.FullName)\RoamingState", "$($_.FullName)\TempState"
            )
        }
}

# Temp + Recent + Cache
Remove-Paths -Paths @(
    "$TargetProfile\AppData\Local\Temp",
    "$TargetProfile\AppData\Roaming\Microsoft\Windows\Recent",
    "$TargetProfile\AppData\Local\Microsoft\Windows\Explorer",
    "$TargetProfile\AppData\Local\Microsoft\Windows\INetCache",
    "$TargetProfile\AppData\Local\Microsoft\Windows\WebCache",
    "$TargetProfile\AppData\Local\CrashDumps"
)

# Reset clipboard history
try {
    Set-ItemProperty -Path "HKCU:\Software\Microsoft\Clipboard" `
        -Name "EnableClipboardHistory" -Value 0 -ErrorAction SilentlyContinue
} catch {}

# MRUs
$MRUKeys = @(
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RecentDocs",
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\RunMRU",
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\TypedPaths",
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\ComDlg32\OpenSavePidlMRU",
    "HKCU:\Software\Microsoft\Windows\CurrentVersion\Explorer\ComDlg32\LastVisitedPidlMRU"
)
foreach ($k in $MRUKeys) { Remove-RegistryKey -KeyPath $k }

Write-Host " [OK] Higiene complementar concluída." -ForegroundColor Green

# ==============================================================================
# VERIFICAÇÃO FINAL
# ==============================================================================

Write-Host ""
Write-Host "[VERIFICAÇÃO] Conferindo se tudo foi removido..." -ForegroundColor Yellow

$Checks = @{
    "Chrome User Data\Default" = "$TargetProfile\AppData\Local\Google\Chrome\User Data\Default"
    "Chrome Bookmarks"         = "$TargetProfile\AppData\Local\Google\Chrome\User Data\Default\Bookmarks"
    "Edge User Data\Default"   = "$TargetProfile\AppData\Local\Microsoft\Edge\User Data\Default"
    "Firefox Profiles"         = "$TargetProfile\AppData\Roaming\Mozilla\Firefox\Profiles"
    "Office Identity (reg)"    = "HKCU:\Software\Microsoft\Office\16.0\Common\Identity"
    "IdentityCRL StoredIDs"    = "HKCU:\Software\Microsoft\IdentityCRL\StoredIdentities"
    "OneDrive Accounts"        = "HKCU:\Software\Microsoft\OneDrive\Accounts"
}

foreach ($check in $Checks.GetEnumerator()) {
    $exists = Test-Path $check.Value
    $color  = if ($exists) { "Red" } else { "Green" }
    $status = if ($exists) { "AINDA EXISTE" } else { "removido" }
    Write-Host ("   {0,-30} : {1}" -f $check.Key, $status) -ForegroundColor $color
}

# ==============================================================================
# FINALIZAÇÃO
# ==============================================================================

Write-Host ""
Write-Host "============================================================" -ForegroundColor Cyan
Write-Host " RESET DE SESSÃO CONCLUÍDO!"                                 -ForegroundColor Green
Write-Host " Usuário          : $TargetUser"                             -ForegroundColor Cyan
Write-Host " Itens removidos  : $Script:TotalDeleted"                    -ForegroundColor Cyan
Write-Host " Falhas           : $Script:TotalFailed"                     -ForegroundColor $(if ($Script:TotalFailed -gt 0) { "Yellow" } else { "Cyan" })
Write-Host " Data/Hora        : $(Get-Date -Format 'dd/MM/yyyy HH:mm:ss')" -ForegroundColor Cyan
Write-Host "============================================================" -ForegroundColor Cyan
Write-Host ""

# ------------------------------------------------------------------------------
# AVISO IMPORTANTE: limitação de privilégio
# ------------------------------------------------------------------------------
Write-Host "AVISO IMPORTANTE — LIMITAÇÃO DE USUÁRIO COMUM:" -ForegroundColor Yellow
Write-Host "  Se o aluno entrou no Windows COM uma Conta Microsoft (e nao apenas" -ForegroundColor Yellow
Write-Host "  conectou ela em apps), essa conta nao pode ser removida sem admin." -ForegroundColor Yellow
Write-Host "  Recomendacao: usar uma conta LOCAL fixa do Windows (ex: 'Aluno')" -ForegroundColor Yellow
Write-Host "  e deixar os alunos logarem em apps INDIVIDUALMENTE com suas contas." -ForegroundColor Yellow
Write-Host ""

if ($Script:TotalFailed -gt 0) {
    Write-Host "Alguns itens nao foram removidos (provavelmente em uso)." -ForegroundColor Yellow
    Write-Host "Reinicie a maquina e execute novamente para garantir." -ForegroundColor Yellow
    Write-Host ""
}

Write-Host "A maquina esta pronta para o proximo aluno." -ForegroundColor White
Write-Host ""
Write-Host "Esta janela fechara automaticamente em 5 segundos..." -ForegroundColor DarkGray
Start-Sleep -Seconds 5
